# ================= NODE + LIBREOFFICE + PYTHON ENV =================
FROM node:24-bullseye

# Set HOME for LibreOffice
ENV HOME=/tmp

# Install system dependencies
RUN apt-get update && apt-get install -y \
    libreoffice \
    libreoffice-writer \
    libreoffice-calc \
    libreoffice-impress \
    libreoffice-l10n-en \
    libreoffice-pdfimport \
    ffmpeg \
    poppler-utils \
    tesseract-ocr \
    imagemagick \
    python3 \
    python3-venv \
    python3-pip \
    fonts-dejavu-core \
    fonts-liberation \
    && rm -rf /var/lib/apt/lists/*

# Pre-create LibreOffice profile to avoid "no export filter" errors
RUN mkdir -p /tmp/libreoffice && soffice --headless --invisible --nologo --convert-to pdf --outdir /tmp/libreoffice /tmp || true

# ================= PYTHON VENV =================
RUN python3 -m venv /opt/venv
ENV PATH="/opt/venv/bin:$PATH"
RUN pip install --upgrade pip
RUN pip install pdf2docx camelot-py[cv] pandas

# ================= NODE APP =================
WORKDIR /app

# Copy package.json and install dependencies
COPY package*.json ./
RUN npm install

# Copy the rest of the application
COPY . .

# Build the Node project
RUN npm run build

# Expose Express port
EXPOSE 10000

# Start the app
CMD ["npm", "start"]
