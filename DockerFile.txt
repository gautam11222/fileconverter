# Use official Node.js image
FROM node:24.8.0-slim

# Install system dependencies
RUN apt-get update && apt-get install -y \
    libreoffice \
    ffmpeg \
    poppler-utils \
    tesseract-ocr \
    python3 \
    python3-pip \
    fonts-dejavu-core \
    fonts-liberation \
    && apt-get clean \
    && rm -rf /var/lib/apt/lists/*

# Install Python packages
RUN pip3 install pdf2docx camelot-py[cv] pandas

# Set working directory
WORKDIR /app

# Copy package files
COPY package*.json ./

# Install Node dependencies
RUN npm install --production

# Copy the rest of your code
COPY . .

# Build your project
RUN npm run build

# Expose port
EXPOSE 10000

# Start the app
CMD ["npm", "start"]
