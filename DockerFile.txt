# Use Node 22 on Debian Bullseye
FROM node:22-bullseye

# Install system dependencies
RUN apt-get update && apt-get install -y \
    libreoffice \
    ffmpeg \
    poppler-utils \
    tesseract-ocr \
    python3 \
    python3-pip \
    fonts-dejavu-core \
    fonts-liberation \
    && apt-get clean \
    && rm -rf /var/lib/apt/lists/*

# Install Python packages
RUN pip3 install pdf2docx camelot-py[cv] pandas

# Set working directory
WORKDIR /app

# Copy package files
COPY package*.json ./

# Install Node dependencies
RUN npm install --legacy-peer-deps

# Copy the rest of the project
COPY . .

# Build the project
RUN npm run build

# Expose port 10000 (Render default or your app port)
EXPOSE 10000

# Start the app
CMD ["npm", "start"]
